\section{Conclusion}

\subsection{Tips}
\begin{frame}[t]{Tips}
  \begin{enumerate}
    \item use something (hopefully mgunit) {\em regularly}
    \item follow the name conventions
    \item inherit from a subclass of MGutTestCase
    \item skipping is useful
  \end{enumerate}
\end{frame}
